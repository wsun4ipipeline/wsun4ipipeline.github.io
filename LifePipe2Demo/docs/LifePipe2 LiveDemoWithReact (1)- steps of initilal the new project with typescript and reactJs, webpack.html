1. Initial the LiveDemoWithReactJs project with below folders 
  2. open windows command line or terminal, initial npm and install webpack, reactJs, react-dom like below
    • npm init
    • npm install -g webpack
    • npm install --save react react-dom @types/react @types/react-dom
  3. continue install typescript, typescript load
    • npm install --save-dev typescript awesome-typescript-loader source-map-loader
  4. add typescript config in root, file name: tsconfig.json
    • {
          "compilerOptions": {
             "outDir": "./dist/build/",
              "sourceMap": true,
              "noImplicitAny": true,
              "module": "commonjs",
              "target": "es5",
              "jsx": "react"
          },
          "include": [
              "./src/**/*"
          ]
      }
  5. add index.tsx in src
    import * as React from "react";
    import * as ReactDOM from "react-dom";
    import { App } from "./components/APP";

    ReactDOM.render(
        <App text="hello" />,
        document.getElementById("root")
    );
  6. add App.tsx in src/components
    • import * as React from "react";

      export interface AppProps { text:string }

      export class App extends React.Component<AppProps, undefined> {
          render() {
              return <h1>Hello: {this.props.text} </h1>;
          }
  7. add index.html in root
    • <!DOCTYPE html>
      <html>
      <head>
          <meta charset="UTF-8" />
          <title>LifePipe2 demo</title>
      </head>
      <body>
          <div id="root"></div>

          <!-- Dependencies -->
          <script src="./node_modules/react/dist/react.js"></script>
          <script src="./node_modules/react-dom/dist/react-dom.js"></script>

          <!-- Main -->
          <script src="./dist/build/bundle.js"></script>
      </body>
      </html>
  8. under project root, add webpack.config.js file for build
    • module.exports = {
          entry: "./src/index.tsx",
          output: {
              filename: "app.bundle.js",
              path: __dirname + "/dist/build"
          },

          // Enable sourcemaps for debugging webpack's output.
          devtool: "source-map",

          resolve: {
              // Add '.ts' and '.tsx' as resolvable extensions.
              extensions: [".ts", ".tsx", ".js", ".json"]
          },

          module: {
              rules: [
                  // All files with a '.ts' or '.tsx' extension will be handled by 'awesome-typescript-loader'.
                  { test: /\.tsx?$/, loader: "awesome-typescript-loader" },

                  // All output '.js' files will have any sourcemaps re-processed by 'source-map-loader'.
                  { enforce: "pre", test: /\.js$/, loader: "source-map-loader" }
              ]
          },

          // When importing a module whose path matches one of the following, just
          // assume a corresponding global variable exists and use that instead.
          // This is important because it allows us to avoid bundling all of our
          // dependencies, which allows browsers to cache those libraries between builds.
          externals: {
              "react": "React",
              "react-dom": "ReactDOM"
          },
      };
  9. run webpack in command line in the root
    • the app.bundle.js and app.bundle.js.map was created in dist/build folder as configed.
